function fn(arr) {
    const sortArr = arr.sort((a,b) => a[0] - b[0]);
    console.log(sortArr);
    let number = 1;
    let end = sortArr[0][1];
    for (let i = 1; i < sortArr.length; i++) {
        if (sortArr[i][0] < end) {
            end !== sortArr[i][0] ? number++ : '';
        }
        end = sortArr[i][1];
    }
    return number;
}
console.log(fn([[0, 30],[5, 10],[15, 20], [10,35], [20, 30]]));

  for (const item of data) {

      for (const m of item.children) {
        result[m] = {}
        dfs(data, result[m], m)

  }
}

  1. 查找一个字符串中最长的回文子串
const longReverseStr = str => {
    if (str.length < 2>) return str;
    const len = str.length;
    let res = str[0];
    const dp = Array.from(Array(len), () => Array(len).fill(0));
    for (let i = 0; i < len; i++) {
        dp[i][i] = true;
    }
    for(let i = 1; i < len; i++) {
        for (let j = 0; j < i; j++) {
            if (j - i == 1 && str[i] === str[j]) {
                dp[i][j] = true;
            } else if(str[j] === str[j] && str[i] === str[j]) {
                dp[i][j] = true;
            }
            if (dp[i][j] && j - i + 1 > res.length) {
                res = str.substring(i, j + 1);
            }
        }
    }
    return res;
}
