const fn = str => {
    let len = str.length;
    let res = ''
    let aMax = 0,bMax = 0;
    let maxLen = str.length
    for(let i = 0; i < len; i++) {
        for(let j = i; j < len; j++) {
            let s = str.substring(i, j)
            let len = isValid(s);
            if(len.a >= aMax && len.b >= bMax) {
                if(len.a === aMax && len.b === bMax && len.len < maxLen) {
                    res = s;
                    aMax = len.a
                    bMax = len.b
                    maxLen = len.len
                    break;
                }
                res = s;
                aMax = len.a
                bMax = len.b
                maxLen = len.len
            }
        }
    }
    return res;
}
