1. 字符串->字符串
```javascript
var str = '2[a]2[cc3[d]]'  // aaccdddccddd

function dfs(reg, str) {
  return str.replace(reg, (rs, $1, $2)=> {
    let res = '';
    for (let i = 0; i < $1; i++) {
      res = `${res}${$2 + ''}`;
    }
    return res
  })
}
function fn(str) {
  let reg = /(\d)\[(\w*)\]/g;
  let res = str;
  while(reg.exec(res)) {
    res = dfs(reg, res);
  }
  return res;
}

console.log(fn(str))
```
