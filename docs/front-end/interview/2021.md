1. 查找一个字符串中最长的回文子串
```
function huiString(s) {
    var t = s.split('').join('#');
    t = '~' + t + '#';
    var c = 1, e = 0, cs = 0; // c表示错位相互比较
    for(var j = 1; j < t.length - 1; j++, c = 1) {
        console.log(j - c);
        while(t[j + c] === t[j - c]) {
            c++;
        }
        if (c > e) { // c++ 会产生大于e,将c赋予e, cs等于j；
            e = c;
            cs = j;
        }
    }
    var result = t.slice(cs - e + 1, cs + e).replace(/#/g, ''). replace(/~/g, '');
    return result;
}
```
